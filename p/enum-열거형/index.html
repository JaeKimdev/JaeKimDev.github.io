enum Weekday: Int {
    case Monday = 1
    case Tuesday
    case Wednesday
    case Thursday
    case Friday
    case Saturday
    case Sunday

    func name() -> String {
        switch self {
        case .Monday, .Tuesday, .Wednesday, .Thursday, .Friday:
            return "Weekday"

        case .Saturday, .Sunday:
            return "Weekends"
        }
    }
}

print(Weekday.Monday) // Monday
print(Weekday.Sunday.rawValue) // 7
print(Weekday.Monday.name()) // Weekday
print(Weekday.Monday.name) // (Function)
